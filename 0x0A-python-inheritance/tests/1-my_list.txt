=============================
The ``1-my_list`` module
=============================

Using ``MyList``
---------------------

Import the class:

    >>> MyList = __import__('1-my_list').MyList

Now test it:

    >>> my_list = MyList()
    >>> my_list.append(10)
    >>> my_list.append(7)
    >>> my_list.append(14)
    >>> my_list.append(21)
    >>> my_list.append(3)

    >>> my_list.print_sorted()
    [3, 7, 10, 14, 21]

    >>> print(my_list)
    [10, 7, 14, 21, 3]

    >>> my_list.print_sorted() == sorted(my_list)
    False  # Values are different now

    >>> my_list = MyList("world")
    >>> my_list
    ['w', 'o', 'r', 'l', 'd']

    >>> my_list.print_sorted()
    ['d', 'l', 'o', 'r', 'w']

    >>> my_list = MyList()
    >>> print(my_list)
    []
    >>> my_list.print_sorted()
    []

    >>> my_list.append(8)
    >>> my_list.append(17)
    >>> print(my_list)
    [8, 17]
    >>> my_list.print_sorted()
    [8, 17]

    >>> my_list.append(-5)
    >>> my_list.print_sorted()
    [-5, 8, 17]

    >>> my_list = MyList([4, 8, 12])
    >>> sorted_list = my_list.print_sorted()
    >>> print(my_list)
    [4, 8, 12]
    >>> my_list == sorted_list
    False
