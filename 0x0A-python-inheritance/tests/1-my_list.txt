=============================
The ``1-my_list`` module
=============================

Using ``MyList``
---------------------

Import the class:

    >>> MyList = __import__('1-my_list').MyList

    >>> my_list = MyList()
    >>> my_list.append(1)
    >>> my_list.append(4)
    >>> my_list.append(2)
    >>> my_list.append(3)
    >>> my_list.append(5)

    >>> my_list.print_sorted()
    [1, 2, 3, 4, 5]

    >>> print(my_list)
    [1, 4, 2, 3, 5]

    >>> my_list.print_sorted() == sorted(my_list)
    True

    >>> my_list = MyList("hello")
    >>> my_list
    ['h', 'e', 'l', 'l', 'o']

    >>> my_list.print_sorted()
    ['e', 'h', 'l', 'l', 'o']

    >>> my_list = MyList()
    >>> print(my_list)
    []
    >>> my_list.print_sorted()
    []

    >>> my_list.append(3)
    >>> my_list.append(2)
    >>> print(my_list)
    [3, 2]
    >>> my_list.print_sorted()
    [2, 3]

    >>> my_list.append(-12)
    >>> my_list.print_sorted()
    [-12, 2, 3]

    >>> my_list = MyList([1, 2, 3])
    >>> sorted_list = my_list.print_sorted()
    >>> print(my_list)
    [1, 2, 3]
    >>> my_list == sorted_list
    True
